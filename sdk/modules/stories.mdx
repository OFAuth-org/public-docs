## Overview

The OFAuth SDK does not expose dedicated story CRUD endpoints. OnlyFans stories are ephemeral content that auto-expire, and the platform's API surface for stories is limited. However, you can access **story analytics** through the analytics namespace, and reach any story endpoint directly via the **proxy method**.

## Story Analytics

The SDK includes analytics endpoints that return story performance data: view counts, tips, likes, and top-performing stories.

### Story Analytics Chart

<CodeGroup>
```typescript TypeScript
import { createOFAuthClient } from '@ofauth/onlyfans-sdk';

const client = createOFAuthClient({
  apiKey: 'your-api-key',
  connectionId: 'conn_xxx',
});

try {
  const chart = await client.access.analytics.stories.getChart({
    startDate: '2024-01-01',
    endDate: '2024-01-31',
    by: 'viewers',
  });

  console.log('Total views:', chart.total);
  console.log('Change:', chart.delta);
  for (const point of chart.chart) {
    console.log(`${point.date}: ${point.count}`);
  }
} catch (error) {
  console.error('Failed to get story analytics:', error);
}
```

```python Python
from onlyfans_sdk import OFAuthClient, analytics

client = OFAuthClient(
    api_key="your-api-key",
    connection_id="conn_xxx",
)

try:
    chart = analytics.list_stories_charts(
        client,
        start_date="2024-01-01",
        end_date="2024-01-31",
        by="viewers",
    )

    print("Total views:", chart["total"])
    print("Change:", chart["delta"])
    for point in chart["chart"]:
        print(f"{point['date']}: {point['count']}")
except Exception as e:
    print("Failed to get story analytics:", e)
```

```go Go
ctx := context.Background()
client := ofauth.NewClient("your-api-key").
    WithConnectionID("conn_xxx")

chart, err := client.GetAccessAnalyticsStoriesChart(ctx, &ofauth.RequestOptions{
    Query: url.Values{
        "startDate": []string{"2024-01-01"},
        "endDate":   []string{"2024-01-31"},
        "by":        []string{"viewers"},
    },
})
if err != nil {
    log.Fatal(err)
}

fmt.Printf("Total views: %d\n", chart.Total)
fmt.Printf("Change: %d\n", chart.Delta)
for _, point := range chart.Chart {
    fmt.Printf("%s: %d\n", point.Date, point.Count)
}
```

```csharp C#
using OFAuth;

var client = new OFAuthClient("your-api-key")
    .WithConnectionId("conn_xxx");

try
{
    var chart = await client.GetAsync<StoriesChartResponse>(
        "/v2/access/analytics/stories/chart",
        query: new Dictionary<string, string?>
        {
            ["startDate"] = "2024-01-01",
            ["endDate"] = "2024-01-31",
            ["by"] = "viewers"
        }
    );

    Console.WriteLine($"Total views: {chart.Total}");
    Console.WriteLine($"Change: {chart.Delta}");
}
catch (OFAuthException ex)
{
    Console.WriteLine($"Error: {ex.Status} {ex.Message}");
}
```
</CodeGroup>

### Top Stories

Retrieve your highest-performing stories by a given metric:

<CodeGroup>
```typescript TypeScript
const topStories = await client.access.analytics.stories.getTop({
  startDate: '2024-01-01',
  endDate: '2024-01-31',
  by: 'viewers',
});

for (const story of topStories.items) {
  console.log(`Story ${story.id}: ${story.viewersCount} viewers, ${story.likesCount} likes`);
}
```

```python Python
top_stories = analytics.list_stories_tops(
    client,
    start_date="2024-01-01",
    end_date="2024-01-31",
    by="viewers",
)

for story in top_stories["items"]:
    print(f"Story {story['id']}: {story['viewersCount']} viewers, {story['likesCount']} likes")
```

```go Go
topStories, err := client.GetAccessAnalyticsStoriesTop(ctx, &ofauth.RequestOptions{
    Query: url.Values{
        "startDate": []string{"2024-01-01"},
        "endDate":   []string{"2024-01-31"},
        "by":        []string{"viewers"},
    },
})
if err != nil {
    log.Fatal(err)
}

for _, story := range topStories.Items {
    fmt.Printf("Story %d: %d viewers\n", story.Id, story.ViewersCount)
}
```

```csharp C#
var topStories = await client.GetAsync<StoriesTopResponse>(
    "/v2/access/analytics/stories/top",
    query: new Dictionary<string, string?>
    {
        ["startDate"] = "2024-01-01",
        ["endDate"] = "2024-01-31",
        ["by"] = "viewers"
    }
);

foreach (var story in topStories.Items)
{
    Console.WriteLine($"Story {story.Id}: {story.ViewersCount} viewers");
}
```
</CodeGroup>

## Accessing Story Endpoints via Proxy

For any OnlyFans story endpoint not covered by the SDK's typed methods, use the proxy to call the API directly:

<CodeGroup>
```typescript TypeScript
// Fetch stories for the authenticated user
const stories = await client.proxy({
  path: '/stories',
  method: 'GET',
  connectionId: 'conn_xxx',
});

console.log('Stories:', stories);

// Delete a story
await client.proxy({
  path: '/stories/123456',
  method: 'DELETE',
  connectionId: 'conn_xxx',
});
```

```python Python
# Fetch stories for the authenticated user
stories = client.proxy(
    "/stories",
    connection_id="conn_xxx",
)

print("Stories:", stories)

# Delete a story
client.proxy(
    "/stories/123456",
    method="DELETE",
    connection_id="conn_xxx",
)
```

```go Go
// Fetch stories for the authenticated user
body, err := client.Request(ctx, "GET", "/v2/access/proxy/stories", &ofauth.RequestOptions{
    ConnectionID: "conn_xxx",
})
if err != nil {
    log.Fatal(err)
}
fmt.Println("Stories:", string(body))

// Delete a story
_, err = client.Request(ctx, "DELETE", "/v2/access/proxy/stories/123456", &ofauth.RequestOptions{
    ConnectionID: "conn_xxx",
})
```

```csharp C#
// Fetch stories for the authenticated user
var stories = await client.GetAsync<List<Dictionary<string, object>>>(
    "/v2/access/proxy/stories"
);

// Delete a story
await client.DeleteAsync<object>("/v2/access/proxy/stories/123456");
```
</CodeGroup>

<Tip>
  The proxy method lets you call **any** OnlyFans API endpoint. This is the recommended approach for story creation, deletion, and other operations not yet exposed as typed SDK methods.
</Tip>
